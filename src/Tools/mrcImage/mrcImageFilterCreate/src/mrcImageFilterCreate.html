    [-sz                 Sz                  (1         ).as(Integer             ) ] :Optional  :FilterSize: z
    [-v[erbose]                                                                    ] :Optional  :Verbose

    [-M[ode]             Mode                (0         ).as(Integer             ) ] :Optional  :Mode: Filter shape
    [-m[ode]             mode                (2         ).as(Integer             ) ] :Optional  :mode: output mrc
