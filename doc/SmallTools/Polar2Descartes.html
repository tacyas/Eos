    [-zaxis              zx                  (0.0       ).as(Real                )

                         zz                  (1.0       ).as(Real                ) ] :Optional  :z-axis
    [-c[onfig]           configFile          (NULL      ).as(inFile              ) ] :Optional  :ConfigurationFile
    [-m[ode]             mode                (0         ).as(Integer             ) ] :Optional  :Mode
